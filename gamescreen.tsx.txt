import React, { useState, useEffect, useRef } from 'react';
import { Question } from '../types';
import { Button } from './Button';
import { generateSpeechBase64 } from '../services/geminiService';

interface GameScreenProps {
  questions: Question[];
  onFinish: (score: number) => void;
}

function decode(base64: string) {
  const binaryString = atob(base64);
  const len = binaryString.length;
  const bytes = new Uint8Array(len);
  for (let i = 0; i < len; i++) {
    bytes[i] = binaryString.charCodeAt(i);
  }
  return bytes;
}

async function decodeAudioData(
  data: Uint8Array,
  ctx: AudioContext,
  sampleRate: number,
  numChannels: number,
): Promise<AudioBuffer> {
  const dataInt16 = new Int16Array(data.buffer);
  const frameCount = dataInt16.length / numChannels;
  const buffer = ctx.createBuffer(numChannels, frameCount, sampleRate);

  for (let channel = 0; channel < numChannels; channel++) {
    const channelData = buffer.getChannelData(channel);
    for (let i = 0; i < frameCount; i++) {
      channelData[i] = dataInt16[i * numChannels + channel] / 32768.0;
    }
  }
  return buffer;
}

export const GameScreen: React.FC<GameScreenProps> = ({ questions, onFinish }) => {
  const [currentIndex, setCurrentIndex] = useState(0);
  const [score, setScore] = useState(0);
  const [selectedAnswer, setSelectedAnswer] = useState<string | null>(null);
  const [isAnswerRevealed, setIsAnswerRevealed] = useState(false);

  const [isLoadingAudio, setIsLoadingAudio] = useState(false);
  const [isPlayingAudio, setIsPlayingAudio] = useState(false);
  const audioSourceRef = useRef<AudioBufferSourceNode | null>(null);

  const currentQuestion = questions[currentIndex];
  const totalQuestions = questions.length;
  const progress = ((currentIndex) / totalQuestions) * 100;

  useEffect(() => {
    if (audioSourceRef.current) {
      audioSourceRef.current.stop();
      audioSourceRef.current = null;
    }
    setIsPlayingAudio(false);
    setIsLoadingAudio(false);
  }, [currentIndex]);

  const handleReadQuestion = async () => {
    if (isLoadingAudio || isPlayingAudio) return;

    setIsLoadingAudio(true);
    try {
      const base64Audio = await generateSpeechBase64(currentQuestion.text);
      if (!base64Audio) throw new Error("لم يتم استلام بيانات صوتية");

      const AudioContextClass = window.AudioContext || (window as any).webkitAudioContext;
      const outputAudioContext = new AudioContextClass({ sampleRate: 24000 });
      const outputNode = outputAudioContext.createGain();

      const audioBuffer = await decodeAudioData(
        decode(base64Audio),
        outputAudioContext,
        24000,
        1
      );

      const source = outputAudioContext.createBufferSource();
      source.buffer = audioBuffer;
      source.connect(outputNode);
      outputNode.connect(outputAudioContext.destination);

      source.onended = () => {
        setIsPlayingAudio(false);
        audioSourceRef.current = null;
      };

      audioSourceRef.current = source;
      setIsPlayingAudio(true);
      source.start();
    } catch (error) {
      console.error("Failed to play audio:", error);
    } finally {
      setIsLoadingAudio(false);
    }
  };

  const handleAnswerClick = (answer: string) => {
    if (isAnswerRevealed) return;

    setSelectedAnswer(answer);
    setIsAnswerRevealed(true);

    const isCorrect = answer === currentQuestion.correctAnswer;
    if (isCorrect) {
      setScore((prev) => prev + 1);
    }

    setTimeout(() => {
      if (currentIndex < totalQuestions - 1) {
        setCurrentIndex((prev) => prev + 1);
        setSelectedAnswer(null);
        setIsAnswerRevealed(false);
      } else {
        const finalScore = score + (isCorrect ? 1 : 0);
        onFinish(finalScore);
      }
    }, 1500);
  };

  if (!currentQuestion) return null;

  return (
    <div className="flex flex-col h-full w-full max-w-2xl mx-auto py-4">
      <div className="mb-8 space-y-4">
        <div className="flex justify-between items-center text-gray-600 font-bold px-2">
          <span>السؤال {currentIndex + 1} من {totalQuestions}</span>
          <span className="bg-indigo-100 text-indigo-800 px-4 py-1 rounded-full">النقاط: {score}</span>
        </div>
        <div className="w-full bg-gray-200 rounded-full h-3 overflow-hidden">
          <div 
            className="bg-gradient-to-r from-indigo-500 to-purple-500 h-3 rounded-full transition-all duration-500 ease-out" 
            style={{ width: `${progress}%` }}
          ></div>
        </div>
      </div>

      <div className="bg-white rounded-3xl shadow-xl p-8 mb-8 transform transition-all relative">
        <div className="absolute top-4 left-4">
          <button
            onClick={handleReadQuestion}
            disabled={isLoadingAudio || isPlayingAudio}
            className={`p-3 rounded-full transition-all shadow-sm flex items-center justify-center
              ${isLoadingAudio ? 'bg-gray-100 text-gray-400 cursor-wait' : 
                isPlayingAudio ? 'bg-indigo-100 text-indigo-600 animate-pulse' : 
                'bg-gray-50 text-gray-500 hover:bg-indigo-50 hover:text-indigo-600'
              }`}
            title="قراءة السؤال بصوت عالٍ"
          >
            {isLoadingAudio ? (
              <svg className="animate-spin h-6 w-6" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
                <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
                <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
              </svg>
            ) : (
              <svg className="h-6 w-6" fill="none" viewBox="0 0 24 24" stroke="currentColor" strokeWidth={2}>
                <path strokeLinecap="round" strokeLinejoin="round" d="M15.536 8.464a5 5 0 010 7.072m2.828-9.9a9 9 0 010 12.728M12 18.5V5.5L7 10.5H4v3h3l5 5z" />
              </svg>
            )}
          </button>
        </div>
        <h2 className="text-3xl font-bold text-gray-800 leading-tight text-center mt-4">
          {currentQuestion.text}
        </h2>
      </div>

      <div className="grid grid-cols-1 sm:grid-cols-2 gap-4 flex-grow">
        {currentQuestion.options.map((option, index) => {
          let variant: 'secondary' | 'success' | 'danger' = 'secondary';
          let buttonClass = 'min-h-[80px] text-xl transition-all duration-300';
          
          if (isAnswerRevealed) {
            if (option === currentQuestion.correctAnswer) {
              variant = 'success';
              buttonClass += ' scale-[1.02] shadow-lg ring-4 ring-emerald-200';
            } else if (option === selectedAnswer) {
              variant = 'danger';
              buttonClass += ' opacity-50';
            } else {
               buttonClass += ' opacity-50';
            }
          } else if (selectedAnswer === option) {
             buttonClass += ' scale-95';
          }

          return (
            <Button
              key={index}
              variant={variant}
              fullWidth
              className={buttonClass}
              onClick={() => handleAnswerClick(option)}
              disabled={isAnswerRevealed}
            >
              {option}
            </Button>
          );
        })}
      </div>
    </div>
  );
};
